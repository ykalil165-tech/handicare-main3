\documentclass[border=10pt]{standalone}
\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}
\usepackage{tikz}
\usepackage{tikz-uml}
\usepackage{lmodern}

\begin{document}
\begin{tikzpicture}

\begin{umlseqdiag}
\umlobject[class=Client]{Frontend}
\umlobject[class=REST]{AuthController}
\umlobject[class=Service]{AuthService}
\umlobject[class=Repository]{AdminUserRepo}
\umlobject[class=Repository]{AppUserRepo}

\begin{umlcall}[op={POST /api/auth/register(fullName, email, password, preferredNeed)}, return={AuthResponse(token, user)}]{Frontend}{AuthController}
  \begin{umlcall}[op={register(RegisterRequest)}, return={AuthResponse}]{AuthController}{AuthService}
    \begin{umlcall}[op={findByEmailIgnoreCase(email)}, return={Optional<AdminUser>}]{AuthService}{AdminUserRepo}
    \end{umlcall}
    \begin{umlcall}[op={findByEmailIgnoreCase(email)}, return={Optional<AppUser>}]{AuthService}{AppUserRepo}
    \end{umlcall}
    \begin{umlfragment}[type=alt, label={email already exists}, inner xsep=2]
      \begin{umlcallself}[op={throw IllegalArgumentException("Email already used")}]{AuthService}
      \end{umlcallself}
    \umlfpart[else: email available]
      \begin{umlcallself}[op={hashPassword(password)}]{AuthService}
      \end{umlcallself}
      \begin{umlcallself}[op={new AppUser(fullName, email, preferredNeed, hash)}]{AuthService}
      \end{umlcallself}
      \begin{umlcall}[op={save(AppUser)}, return={AppUser (with id)}]{AuthService}{AppUserRepo}
      \end{umlcall}
      \begin{umlcallself}[op={createToken()}]{AuthService}
      \end{umlcallself}
      \begin{umlcallself}[op={sessions.put(token, session)}]{AuthService}
      \end{umlcallself}
    \end{umlfragment}
  \end{umlcall}
\end{umlcall}

\end{umlseqdiag}

\end{tikzpicture}
\end{document}
