% ===== Service Classes =====

\umlclass[type=abstract, x=4, y=4]{AuthService}{
  - sessions : Map<String, SessionAccount>
}{
  + login(AuthRequest) : AuthResponse \\
  + register(RegisterRequest) : AuthResponse \\
  + requireUser(authHeader) : AppUser \\
  + requireAdmin(authHeader) : AdminUser \\

% ===== Relationships =====
